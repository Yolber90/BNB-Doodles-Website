        $(document).ready(function () {

            let link = "https://raw.githubusercontent.com/freeCodeCamp/ProjectReferenceData/master/GDP-data.json";// Calibration Link

            $.get("database", function (data) {

                // Variable Declerations
                const objData = JSON.parse(data); //convert data into an obj
                const pups = objData.puppies //short cont for for data.puppies
                const dataLen = Object.keys(pups).length // get the obj length
                var today = new Date();
                var born = new Date(pups[0].DOB)
                var daySince = today.getTime() - born.getTime()
                var diff = daySince / (1000 * 3600 * 24);
                var finalDiff = Math.round(diff);
                let i = 0;


                //For Loop, load a id 'posting' for each puppy
                for (i; i < dataLen; i++) {
                    // append a div with id #posting under div with id #listing
                    $("#listing").append(
                        "<div id='posting' class= img" + [i] + "></div>"
                    )
                    //Add image to #listing
                    $(".img" + [i]).append(
                        pups[i].thumbnail
                    )
                    $(".img" + [i]).append(
                        "<div id='info' class=info" + [i] + ">" +
                        "<ul id=ul-list" + [i] + "></ul>" + "</div>"
                    )
                    $("#ul-list" + [i]).append(
                        "<li class='ident-name'><strong>" + pups[i].name + "</strong></li>" +
                        "<li id='ident'>Coat: " + pups[i].coat + "</li>" +
                        "<li id='ident'>DOB:  " + pups[i].DOB + "-" + "<br>" + finalDiff + " Days Old</li>"
                    )
                    $(".img" + [i]).click(function () {
                        window.open("https://www.google.com")
                    });
                };
            })

        });
